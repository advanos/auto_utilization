00234     // Set cpu busy circle time.
00235     long busy = total * percent;
00236     // Set cpu idle circle time.
00237     long idle = total - busy;

00253         // Initialize start and stop time.
00254         gettimeofday(&begin, NULL);
00255         start = begin.tv_sec * micro + begin.tv_usec;
00256         stop = start;
00257         // Busy circle.

00260             gettimeofday(&end, NULL);
